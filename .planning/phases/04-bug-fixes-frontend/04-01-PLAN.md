```go
func TestWatcherDetectsCreateThenEdit(t *testing.T) {
    dir := t.TempDir()
    cfgFile := filepath.Join(dir, "config.toml")

    var reloadCount atomic.Int32
    w, err := NewWatcher([]string{cfgFile}, func() error {
        reloadCount.Add(1)
        return nil
    }, 200*time.Millisecond)
    if err != nil {
        t.Fatal(err)
    }
    defer w.Close()
    w.Start()
    time.Sleep(100 * time.Millisecond)

    // Create file (should trigger first reload)
    os.WriteFile(cfgFile, []byte("[defaults]\n"), 0o644)
    time.Sleep(500 * time.Millisecond)

    countAfterCreate := reloadCount.Load()
    if countAfterCreate < 1 {
        t.Fatalf("expected at least 1 reload after create, got %d", countAfterCreate)
    }

    // Edit file -- this is where BUG-03 manifests
    // After dir-to-file transition, edits should still trigger reload
    os.WriteFile(cfgFile, []byte("[defaults]\nimage = \"updated\"\n"), 0o644)
    time.Sleep(500 * time.Millisecond)

    if reloadCount.Load() <= countAfterCreate {
        t.Errorf("expected reload after edit, count stayed at %d", countAfterCreate)
    }
}
```

```go
package session

import (
    "strings"
    "testing"
    "time"

    "github.com/bketelsen/clincus/internal/container"
)

func TestCleanup_TimeoutKeepsContainerRunning(t *testing.T) {
    // This test verifies the backoff PATTERN without waiting for the full
    // 15s timeout. We mock Running() to return false after a few calls,
    // then verify the increasing gaps between calls prove exponential
    // backoff is in effect.
    //
    // Strategy: Use onSequence to return Running=true for 4 calls, then
    // Running=false. With exponential backoff (500ms, 1s, 2s, 4s), the
    // 4 calls take ~7.5s. With old fixed 500ms polling, 4 calls take ~2s.
    // We check elapsed time is > 3s to confirm backoff, and < 12s to
    // confirm we're not waiting for full timeout.

    mock := newSessionMockRunner()
    // Container exists
    mock.on("incus list", mockResponse{
        stdout:   `[{"name":"test-container","status":"Running"}]`,
        exitCode: 0,
    })
    // Override with sequence: Running=true x 4, then Running=false
    // Running() internally calls "incus list" with the container name
    // We need the sequence to apply to the Running() check specifically.
    // The mock matches on substrings, so we use the container name pattern.
    mock.onSequence("incus list",
        mockResponse{stdout: `[{"name":"test-container","status":"Running"}]`, exitCode: 0},
        mockResponse{stdout: `[{"name":"test-container","status":"Running"}]`, exitCode: 0},
        mockResponse{stdout: `[{"name":"test-container","status":"Running"}]`, exitCode: 0},
        mockResponse{stdout: `[{"name":"test-container","status":"Running"}]`, exitCode: 0},
        mockResponse{stdout: `[{"name":"test-container","status":"Stopped"}]`, exitCode: 0},
    )
    // Delete succeeds
    mock.on("incus delete", mockResponse{stdout: "", exitCode: 0})
    defer container.SetRunner(mock)()

    var logged []string
    opts := CleanupOptions{
        ContainerName: "test-container",
        Logger:        func(msg string) { logged = append(logged, msg) },
    }

    start := time.Now()
    err := Cleanup(opts)
    elapsed := time.Since(start)

    if err != nil {
        t.Fatalf("unexpected error: %v", err)
    }

    // With exponential backoff (500ms + 1s + 2s + 4s = 7.5s for 4 waits),
    // elapsed should be > 3s (proving it's NOT 500ms fixed polling).
    // With old code (500ms x 4 = 2s), this assertion would fail.
    if elapsed < 3*time.Second {
        t.Errorf("expected exponential backoff delays (>3s), but completed in %v -- likely still using fixed 500ms polling", elapsed)
    }

    // Should complete well before the 15s timeout
    if elapsed > 12*time.Second {
        t.Errorf("took too long (%v), expected to stop polling after container reports stopped", elapsed)
    }
}

    }

    // Verify delete was called
    deleteCalls := mock.callsContaining("incus delete")
    if len(deleteCalls) == 0 {
        t.Error("expected delete to be called for stopped container")
    }
}

func TestCleanup_TimeoutLogsKeptRunning(t *testing.T) {
    // Separate test for the timeout log message. Uses a container that
    // always reports Running=true. This test WILL wait for the full 15s

        ContainerName: "test-stuck",

    if err != nil {
        t.Fatalf("unexpected error: %v", err)
    }

    // Should log that container is kept running
    found := false
    for _, msg := range logged {
        if strings.Contains(msg, "kept running") || strings.Contains(msg, "keeping it alive") {
            found = true
            break
        }
    }
    if !found {
        t.Errorf("expected 'kept running' log message, got: %v", logged)
    }
}
```

In `internal/session/cleanup.go`:

2. Replace the polling loop (lines 80-87):
```go
running := true
for i := 0; i < 10; i++ {
    time.Sleep(500 * time.Millisecond)
    running, _ = mgr.Running()
    if !running {
        break
    }
}
```

With exponential backoff using labeled loop (to avoid pitfall 1 -- select break vs for break):
```go
ctx, cancel := context.WithTimeout(context.Background(), 15*time.Second)
defer cancel()

running := true
backoff := 500 * time.Millisecond
maxBackoff := 4 * time.Second

loop:
for running {
    select {
    case <-ctx.Done():
        // Timeout -- keep container running (non-destructive)
        opts.Logger("Container still running after timeout, keeping it alive")
        break loop
    case <-time.After(backoff):
        running, _ = mgr.Running()
        if backoff < maxBackoff {
            backoff *= 2
        }
    }
}
```

Note: The `break loop` breaks out of the `for` loop, not just the `select`. The labeled loop pattern is the standard Go idiom for this.
